# Primitives

Primitives are low-level components that provide the building blocks for creating custom components with consistent styling and behavior.

## Import

```tsx
import {
  Primitive,
  AccessibleIcon,
  Arrow,
  FocusScope,
  RovingFocusGroup
} from '@radix-ui/themes-native';
```

## Primitive

The base primitive component that all other components are built on.

### Usage

```tsx
import { Primitive } from '@radix-ui/themes-native';

<Primitive.view
  style={{ padding: 16 }}
  onPress={() => console.log('Pressed')}
>
  <Primitive.text>Hello World</Primitive.text>
</Primitive.view>
```

### Props

<PropTable>
  <Prop name="asChild" type="boolean" default="false">
    Merge props onto child element instead of rendering a wrapper.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles.
  </Prop>
</PropTable>

## AccessibleIcon

Makes icon components accessible to screen readers.

### Usage

```tsx
import { AccessibleIcon } from '@radix-ui/themes-native';

<AccessibleIcon label="Search">
  <Icon name="search" />
</AccessibleIcon>
```

### Props

<PropTable>
  <Prop name="children" type="ReactNode" required>
    Icon element.
  </Prop>
  <Prop name="label" type="string" required>
    Accessible label for the icon.
  </Prop>
</PropTable>

## Arrow

Arrow element for tooltips, popovers, and other floating elements.

### Usage

```tsx
import { Arrow } from '@radix-ui/themes-native';

<Popover.Content>
  <Arrow />
  <Text>Popover content</Text>
</Popover.Content>
```

### Props

<PropTable>
  <Prop name="width" type="number" default="10">
    Arrow width.
  </Prop>
  <Prop name="height" type="number" default="5">
    Arrow height.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles.
  </Prop>
</PropTable>

## FocusScope

Manages focus within a container, useful for modals and dialogs.

### Usage

```tsx
import { FocusScope } from '@radix-ui/themes-native';

<FocusScope trapped>
  <TextField placeholder="Focus is trapped here" />
  <Button>Submit</Button>
</FocusScope>
```

### Props

<PropTable>
  <Prop name="children" type="ReactNode">
    Content to manage focus for.
  </Prop>
  <Prop name="trapped" type="boolean" default="false">
    Whether focus is trapped within the scope.
  </Prop>
  <Prop name="onMountAutoFocus" type="(event: Event) => void">
    Called when focus scope mounts.
  </Prop>
  <Prop name="onUnmountAutoFocus" type="(event: Event) => void">
    Called when focus scope unmounts.
  </Prop>
</PropTable>

## RovingFocusGroup

Manages focus navigation between items in a group (e.g., radio group, toolbar).

### Usage

```tsx
import { RovingFocusGroup } from '@radix-ui/themes-native';

<RovingFocusGroup>
  <Button>First</Button>
  <Button>Second</Button>
  <Button>Third</Button>
</RovingFocusGroup>
```

### Props

<PropTable>
  <Prop name="children" type="ReactNode">
    Items to navigate between.
  </Prop>
  <Prop name="orientation" type="'horizontal' | 'vertical'">
    Navigation direction.
  </Prop>
  <Prop name="loop" type="boolean" default="false">
    Whether navigation loops around.
  </Prop>
  <Prop name="currentTabStopId" type="string">
    Controlled current focus ID.
  </Prop>
  <Prop name="onCurrentTabStopIdChange" type="(id: string) => void">
    Callback when focus changes.
  </Prop>
</PropTable>

## Examples

### Custom Button

```tsx
import { Primitive } from '@radix-ui/themes-native';

function CustomButton({ children, variant = 'solid', ...props }) {
  const variantStyles = {
    solid: { backgroundColor: 'blue' },
    soft: { backgroundColor: 'lightblue' },
    outline: { borderWidth: 1, borderColor: 'blue' }
  };

  return (
    <Primitive.view
      role="button"
      style={[
        { padding: 12, borderRadius: 8 },
        variantStyles[variant]
      ]}
      {...props}
    >
      <Primitive.text>{children}</Primitive.text>
    </Primitive.view>
  );
}
```

### Accessible Icon Button

```tsx
import { AccessibleIcon, IconButton } from '@radix-ui/themes-native';

function SearchButton() {
  return (
    <IconButton>
      <AccessibleIcon label="Search">
        <Icon name="search" />
      </AccessibleIcon>
    </IconButton>
  );
}
```

### Focus Trapped Modal

```tsx
import { FocusScope, Dialog } from '@radix-ui/themes-native';

function Modal({ open, onClose }) {
  return (
    <Dialog.Root open={open} onOpenChange={onClose}>
      <Dialog.Content>
        <FocusScope trapped>
          <Dialog.Title>Modal Title</Dialog.Title>
          <Dialog.Description>
            Focus is trapped within this modal.
          </Dialog.Description>
          <Button onPress={onClose}>Close</Button>
        </FocusScope>
      </Dialog.Content>
    </Dialog.Root>
  );
}
```

### Keyboard Navigation

```tsx
import { RovingFocusGroup, Button } from '@radix-ui/themes-native';

function Toolbar() {
  return (
    <RovingFocusGroup orientation="horizontal" loop>
      <Flex gap={2}>
        <Button variant="soft">
          <Icon name="bold" />
        </Button>
        <Button variant="soft">
          <Icon name="italic" />
        </Button>
        <Button variant="soft">
          <Icon name="underline" />
        </Button>
      </Flex>
    </RovingFocusGroup>
  );
}
```

### Custom Popover

```tsx
import { Primitive, Arrow } from '@radix-ui/themes-native';

function CustomPopover({ content, children }) {
  const [open, setOpen] = useState(false);

  return (
    <>
      <Primitive.view onPress={() => setOpen(true)}>
        {children}
      </Primitive.view>

      {open && (
        <Primitive.view style={{ position: 'absolute', top: 40 }}>
          <Arrow fill="white" />
          <Primitive.view style={{ padding: 12, backgroundColor: 'white' }}>
            <Primitive.text>{content}</Primitive.text>
          </Primitive.view>
        </Primitive.view>
      )}
    </>
  );
}
```
