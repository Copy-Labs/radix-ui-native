# ThemeProvider

The `ThemeProvider` component provides theme context to all child components. It should wrap your entire application.

## Import

```tsx
import { ThemeProvider } from '@radix-ui/themes-native';
```

## Usage

```tsx
import { ThemeProvider } from '@radix-ui/themes-native';

export default function App() {
  return (
    <ThemeProvider>
      {/* Your app content */}
    </ThemeProvider>
  );
}
```

## Props

<PropTable>
  <Prop name="children" type="ReactNode" required>
    Child components that will have access to the theme context.
  </Prop>
  <Prop name="mode" type="'light' | 'dark' | undefined" default="undefined">
    Initial theme mode. When undefined, follows the device's color scheme.
  </Prop>
  <Prop name="forcedMode" type="'light' | 'dark' | undefined" default="undefined">
    Force a specific mode, overriding device settings.
  </Prop>
  <Prop name="accentColor" type="string" default="'indigo'">
    Accent color for components. See available colors below.
  </Prop>
  <Prop name="grayColor" type="string" default="'gray'">
    Gray scale variant. See available variants below.
  </Prop>
  <Prop name="radius" type="string" default="'medium'">
    Border radius scale. Options: 'none', 'small', 'medium', 'large', 'full'.
  </Prop>
  <Prop name="scaling" type="number" default="1">
    Scale factor for all sizes. Use values like 0.9 for smaller, 1.1 for larger.
  </Prop>
  <Prop name="onModeChange" type="(mode: ThemeMode) => void">
    Callback when theme mode changes.
  </Prop>
</PropTable>

## Examples

### Basic Setup

```tsx
<ThemeProvider>
  <App />
</ThemeProvider>
```

### With Custom Accent Color

```tsx
<ThemeProvider accentColor="blue">
  <App />
</ThemeProvider>
```

### With Dark Mode

```tsx
<ThemeProvider mode="dark">
  <App />
</ThemeProvider>
```

### Full Configuration

```tsx
<ThemeProvider
  mode="light"
  accentColor="indigo"
  grayColor="mauve"
  radius="medium"
  scaling={1}
  onModeChange={(mode) => console.log('Mode changed:', mode)}
>
  <App />
</ThemeProvider>
```

### With Mode Toggle

```tsx
function AppWithTheme() {
  const [mode, setMode] = useState<'light' | 'dark'>('light');

  return (
    <ThemeProvider
      mode={mode}
      onModeChange={setMode}
    >
      <App />
    </ThemeProvider>
  );
}
```
