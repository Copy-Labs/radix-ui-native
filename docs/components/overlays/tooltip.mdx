# Tooltip

`Tooltip` is an overlay component for displaying contextual information on hover or focus.

## Import

```tsx
import { Tooltip } from '@radix-ui/themes-native';
```

## Usage

```tsx
import { Tooltip, IconButton } from '@radix-ui/themes-native';

<Tooltip content="Copy to clipboard">
  <IconButton>
    <Icon name="copy" />
  </IconButton>
</Tooltip>
```

## Props

<PropTable>
  <Prop name="content" type="ReactNode" required>
    Tooltip content.
  </Prop>
  <Prop name="children" type="ReactNode" required>
    Trigger element.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles.
  </Prop>
  <Prop name="side" type="'top' | 'right' | 'bottom' | 'left'" default="'top'">
    Preferred side to render.
  </Prop>
  <Prop name="sideOffset" type="number" default="4">
    Distance from trigger.
  </Prop>
  <Prop name="align" type="'start' | 'center' | 'end'" default="'center'">
    Alignment relative to trigger.
  </Prop>
  <Prop name="alignOffset" type="number">
    Offset from alignment.
  </Prop>
  <Prop name="delayDuration" type="number" default="400">
    Delay before showing tooltip (ms).
  </Prop>
  <Prop name="skipDelayDuration" type="number" default="300">
    Skip delay when switching between tooltips (ms).
  </Prop>
</PropTable>

## Examples

### Basic Tooltip

```tsx
<Tooltip content="This is a tooltip">
  <Button>Hover me</Button>
</Tooltip>
```

### Positioned Tooltips

```tsx
<Flex gap={4}>
  <Tooltip content="Top tooltip" side="top">
    <Button>Top</Button>
  </Tooltip>

  <Tooltip content="Right tooltip" side="right">
    <Button>Right</Button>
  </Tooltip>

  <Tooltip content="Bottom tooltip" side="bottom">
    <Button>Bottom</Button>
  </Tooltip>

  <Tooltip content="Left tooltip" side="left">
    <Button>Left</Button>
  </Tooltip>
</Flex>
```

### Icon Button Tooltips

```tsx
<Flex gap={2}>
  <Tooltip content="Copy">
    <IconButton>
      <Icon name="copy" />
    </IconButton>
  </Tooltip>

  <Tooltip content="Edit">
    <IconButton>
      <Icon name="pencil" />
    </IconButton>
  </Tooltip>

  <Tooltip content="Delete">
    <IconButton color="red">
      <Icon name="trash" />
    </IconButton>
  </Tooltip>
</Flex>
```

### With Delay

```tsx
<Tooltip content="Shows after 1 second" delayDuration={1000}>
  <Button>Delayed tooltip</Button>
</Tooltip>
```

### Rich Content

```tsx
<Tooltip
  content={
    <Flex direction="column" gap={1}>
      <Text weight="medium">Keyboard Shortcut</Text>
      <Text size={1}>Press ⌘ + C to copy</Text>
    </Flex>
  }
>
  <IconButton>
    <Icon name="keyboard" />
  </IconButton>
</Tooltip>
```

### Form Field Help

```tsx
<Flex direction="column" gap={1}>
  <Flex direction="row" gap={1} align="center">
    <Text size={2} weight="medium">API Key</Text>
    <Tooltip content="Your API key is found in the Developer Settings">
      <IconButton size={1} variant="ghost">
        <Icon name="info" />
      </IconButton>
    </Tooltip>
  </Flex>

</Flex>
```
