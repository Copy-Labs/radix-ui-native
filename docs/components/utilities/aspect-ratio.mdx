# AspectRatio

`AspectRatio` is a utility component for maintaining consistent width-to-height ratios for content.

## Import

```tsx
import { AspectRatio } from '@radix-ui/themes-native';
```

## Usage

```tsx
import { AspectRatio, Image } from '@radix-ui/themes-native';

<AspectRatio ratio={16 / 9}>
  <Image
    source={{ uri: 'https://example.com/image.jpg' }}
    style={{ width: '100%', height: '100%' }}
  />
</AspectRatio>
```

## Props

<PropTable>
  <Prop name="ratio" type="number" required>
    Aspect ratio (width / height). E.g., 16/9, 4/3, 1.
  </Prop>
  <Prop name="children" type="ReactNode">
    Content to maintain aspect ratio for.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles.
  </Prop>
</PropTable>

## Examples

### Common Aspect Ratios

```tsx
<Flex direction="column" gap={3}>
  {/* 16:9 - Video */}
  <AspectRatio ratio={16 / 9}>
    <Image source={{ uri: 'https://example.com/video-thumb.jpg' }} />
  </AspectRatio>

  {/* 4:3 - Standard photo */}
  <AspectRatio ratio={4 / 3}>
    <Image source={{ uri: 'https://example.com/photo.jpg' }} />
  </AspectRatio>

  {/* 1:1 - Square */}
  <AspectRatio ratio={1}>
    <Image source={{ uri: 'https://example.com/square.jpg' }} />
  </AspectRatio>
</Flex>
```

  />
