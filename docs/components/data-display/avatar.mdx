# Avatar

`Avatar` is a data display component for rendering user avatars with image and fallback support.

## Import

```tsx
import { Avatar } from '@radix-ui/themes-native';
```

## Usage

```tsx
import { Avatar } from '@radix-ui/themes-native';

<Avatar
  src="https://example.com/avatar.jpg"
  alt="User avatar"
  fallback="JD"
/>
```

## Props

<PropTable>
  <Prop name="src" type="string">
    Image URL source.
  </Prop>
  <Prop name="alt" type="string">
    Alt text for accessibility.
  </Prop>
  <Prop name="fallback" type="ReactNode">
    Fallback content when image fails to load.
  </Prop>
  <Prop name="size" type="1 | 2 | 3 | 4 | 5 | 6 | 7" default="3">
    Size of the avatar.
  </Prop>
  <Prop name="variant" type="'solid' | 'soft'" default="'soft'">
    Visual style for fallback.
  </Prop>
  <Prop name="color" type="string">
    Accent color for fallback.
  </Prop>
  <Prop name="radius" type="'none' | 'small' | 'medium' | 'large' | 'full'" default="'full'">
    Border radius.
  </Prop>
  <Prop name="highContrast" type="boolean" default="false">
    Uses high contrast colors.
  </Prop>
</PropTable>
