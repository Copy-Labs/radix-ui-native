# Spinner

`Spinner` is a data display component for indicating loading states with an animated spinner.

## Import

```tsx
import { Spinner } from '@radix-ui/themes-native';
```

## Usage

```tsx
import { Spinner } from '@radix-ui/themes-native';

<Spinner />
```

## Props

<PropTable>
  <Prop name="size" type="1 | 2 | 3" default="2">
    Size of the spinner.
  </Prop>
  <Prop name="color" type="string" default="'accent'">
    Color of the spinner.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles.
  </Prop>
</PropTable>

### Conditional Loading

```tsx
function DataLoader() {
  const [loading, setLoading] = useState(true);
  const [data, setData] = useState(null);

  useEffect(() => {
    fetchData().then(result => {
      setData(result);
      setLoading(false);
    });
  }, []);

  if (loading) {
    return (
      <Flex flex={1} justify="center" align="center">
        <Spinner size={3} />
      </Flex>
    );
  }

  return <DataDisplay data={data} />;
}
```

### Card Loading State

```tsx
<Card>
  {loading ? (
    <Flex direction="column" gap={3} align="center" py={4}>
      <Spinner size={2} />
      <Text size={2} color="gray">Loading content...</Text>
    </Flex>
  ) : (
    <CardContent />
  )}
</Card>
```
