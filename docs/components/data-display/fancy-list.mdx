# FancyList

`FancyList` is an enhanced list component for rendering collections with advanced features like dividers, icons, and interactive items.

## Import

```tsx
import { FancyList } from '@radix-ui/themes-native';
```

## Usage

```tsx
import { FancyList } from '@radix-ui/themes-native';

<FancyList>
  <FancyList.Item>First item</FancyList.Item>
  <FancyList.Item>Second item</FancyList.Item>
  <FancyList.Item>Third item</FancyList.Item>
</FancyList>
```

## Props

<PropTable>
  <Prop name="children" type="ReactNode">
    List content.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles.
  </Prop>
  <Prop name="size" type="1 | 2 | 3" default="2">
    Size of the list items.
  </Prop>
  <Prop name="dividers" type="boolean" default="true">
    Show dividers between items.
  </Prop>
  <Prop name="color" type="string">
    Accent color.
  </Prop>
</PropTable>

## Examples

### Basic List

```tsx
<FancyList>
  <FancyList.Item>Apple</FancyList.Item>
  <FancyList.Item>Banana</FancyList.Item>
  <FancyList.Item>Orange</FancyList.Item>
</FancyList>
```

### Without Dividers

```tsx
<FancyList dividers={false}>
  <FancyList.Item>Item 1</FancyList.Item>
  <FancyList.Item>Item 2</FancyList.Item>
  <FancyList.Item>Item 3</FancyList.Item>
</FancyList>
```

### With Icons

```tsx
<FancyList>
  <FancyList.Item icon={<Icon name="home" />}>Home</FancyList.Item>
  <FancyList.Item icon={<Icon name="settings" />}>Settings</FancyList.Item>
  <FancyList.Item icon={<Icon name="user" />}>Profile</FancyList.Item>
</FancyList>
```

### Interactive List

```tsx
<FancyList>
  <FancyList.Item onPress={() => console.log('Pressed item 1')}>
    <Flex direction="row" justify="between" align="center">
      <Text>Notifications</Text>
      <Icon name="chevron-right" />
    </Flex>
  </FancyList.Item>
  <FancyList.Item onPress={() => console.log('Pressed item 2')}>
    <Flex direction="row" justify="between" align="center">
      <Text>Privacy</Text>
      <Icon name="chevron-right" />
    </Flex>
  </FancyList.Item>
</FancyList>
```

### Settings List

```tsx
<Card>
  <Flex direction="column" gap={3}>
    <Heading size={3}>Settings</Heading>
    <FancyList>
      <FancyList.Item>
        <Flex direction="row" justify="between" align="center">
          <Flex direction="row" gap={2} align="center">
            <Icon name="bell" />
            <Text>Notifications</Text>
          </Flex>
          <Switch />
        </Flex>
      </FancyList.Item>
      <FancyList.Item>
        <Flex direction="row" justify="between" align="center">
          <Flex direction="row" gap={2} align="center">
            <Icon name="moon" />
            <Text>Dark Mode</Text>
          </Flex>
          <Switch />
        </Flex>
      </FancyList.Item>
      <FancyList.Item>
        <Flex direction="row" justify="between" align="center">
          <Flex direction="row" gap={2} align="center">
            <Icon name="globe" />
            <Text>Language</Text>
          </Flex>
          <Text color="gray">English</Text>
        </Flex>
      </FancyList.Item>
    </FancyList>
  </Flex>
</Card>
```

### With Badges

```tsx
<FancyList>
  <FancyList.Item>
    <Flex direction="row" justify="between" align="center">
      <Text>Inbox</Text>
      <Badge color="blue">12</Badge>
    </Flex>
  </FancyList.Item>
  <FancyList.Item>
    <Flex direction="row" justify="between" align="center">
      <Text>Drafts</Text>
      <Badge color="gray">3</Badge>
    </Flex>
  </FancyList.Item>
  <FancyList.Item>
    <Flex direction="row" justify="between" align="center">
      <Text>Spam</Text>
      <Badge color="red">5</Badge>
    </Flex>
  </FancyList.Item>
</FancyList>
```
