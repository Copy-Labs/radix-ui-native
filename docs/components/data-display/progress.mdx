# Progress

`Progress` is a data display component for showing completion status of tasks or operations.

## Import

```tsx
import { Progress } from '@radix-ui/themes-native';
```

## Usage

```tsx
import { Progress } from '@radix-ui/themes-native';

<Progress value={60} />
```

## Props

<PropTable>
  <Prop name="value" type="number" default="0">
    Current progress value (0-100).
  </Prop>
  <Prop name="max" type="number" default="100">
    Maximum value.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles.
  </Prop>
  <Prop name="size" type="1 | 2 | 3" default="2">
    Size of the progress bar.
  </Prop>
  <Prop name="color" type="string" default="'accent'">
    Color of the progress bar.
  </Prop>
  <Prop name="radius" type="'none' | 'small' | 'medium' | 'large' | 'full'">
    Border radius override.
  </Prop>
  <Prop name="variant" type="'soft' | 'surface' | 'gradient'" default="'soft'">
    Visual style variant.
  </Prop>
</PropTable>

### Dynamic Progress

```tsx
function UploadProgress() {
  const [progress, setProgress] = useState(0);

  useEffect(() => {
    const interval = setInterval(() => {
      setProgress(prev => {
        if (prev >= 100) {
          clearInterval(interval);
          return 100;
        }
        return prev + 10;
      });
    }, 500);

    return () => clearInterval(interval);
  }, []);

  return (
    <Flex direction="column" gap={2}>
      <Text>Uploading: {progress}%</Text>
      <Progress value={progress} color={progress === 100 ? 'green' : 'blue'} />
    </Flex>
  );
}
```
