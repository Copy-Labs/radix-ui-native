# Badge

`Badge` is a data display component for rendering status indicators and labels.

## Import

```tsx
import { Badge } from '@radix-ui/themes-native';
```

## Usage

```tsx
import { Badge } from '@radix-ui/themes-native';

<Badge color="green">Active</Badge>
```

## Props

<PropTable>
  <Prop name="children" type="ReactNode">
    Badge content.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles.
  </Prop>
  <Prop name="variant" type="'solid' | 'soft' | 'outline'" default="'soft'">
    Visual style variant.
  </Prop>
  <Prop name="size" type="1 | 2 | 3" default="1">
    Size of the badge.
  </Prop>
  <Prop name="color" type="string">
    Accent color.
  </Prop>
  <Prop name="highContrast" type="boolean" default="false">
    Uses high contrast colors.
  </Prop>
  <Prop name="radius" type="'none' | 'small' | 'medium' | 'large' | 'full'">
    Border radius override.
  </Prop>
</PropTable>
