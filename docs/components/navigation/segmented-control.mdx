# SegmentedControl

`SegmentedControl` is a navigation component for selecting between mutually exclusive options, similar to a radio group but with a horizontal button-like appearance.

## Import

```tsx
import { SegmentedControl } from '@radix-ui/themes-native';
```

## Usage

```tsx
import { SegmentedControl } from '@radix-ui/themes-native';

<SegmentedControl.Root defaultValue="day">
  <SegmentedControl.Item value="day">Day</SegmentedControl.Item>
  <SegmentedControl.Item value="week">Week</SegmentedControl.Item>
  <SegmentedControl.Item value="month">Month</SegmentedControl.Item>
</SegmentedControl.Root>
```

## Components

### SegmentedControl.Root

The root container component that manages selection state.

### SegmentedControl.Item

Individual option item.

## Props

### SegmentedControl.Root

<PropTable>
  <Prop name="children" type="ReactNode">
    Segmented control items.
  </Prop>
  <Prop name="defaultValue" type="string">
    Initial selected value (uncontrolled).
  </Prop>
  <Prop name="value" type="string">
    Controlled selected value.
  </Prop>
  <Prop name="onValueChange" type="(value: string) => void">
    Callback when selection changes.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles.
  </Prop>
  <Prop name="size" type="1 | 2 | 3" default="2">
    Size of the control.
  </Prop>
  <Prop name="color" type="string" default="'accent'">
    Color of the selected item.
  </Prop>
</PropTable>

### SegmentedControl.Item

<PropTable>
  <Prop name="value" type="string" required>
    Unique value for this item.
  </Prop>
  <Prop name="children" type="ReactNode">
    Item content.
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether the item is disabled.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles.
  </Prop>
</PropTable>

### Controlled

```tsx
function ControlledSegmentedControl() {
  const [view, setView] = useState('list');

  return (
    <Flex direction="column" gap={3}>
      <SegmentedControl.Root value={view} onValueChange={setView}>
        <SegmentedControl.Item value="list">List</SegmentedControl.Item>
        <SegmentedControl.Item value="grid">Grid</SegmentedControl.Item>
        <SegmentedControl.Item value="table">Table</SegmentedControl.Item>
      </SegmentedControl.Root>

      {view === 'list' && <ListView />}
      {view === 'grid' && <GridView />}
      {view === 'table' && <TableView />}
    </Flex>
  );
}
```

### With Disabled Item
