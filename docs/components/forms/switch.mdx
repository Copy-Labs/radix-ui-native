# Switch

`Switch` is a form component for toggling between on/off states.

## Import

```tsx
import { Switch } from '@radix-ui/themes-native';
```

## Usage

```tsx
import { Switch } from '@radix-ui/themes-native';

<Switch
  checked={enabled}
  onCheckedChange={setEnabled}
/>
```

## Props

<PropTable>
  <Prop name="checked" type="boolean" default="false">
    Whether the switch is on.
  </Prop>
  <Prop name="onCheckedChange" type="(checked: boolean) => void">
    Called when checked state changes.
  </Prop>
  <Prop name="disabled" type="boolean" default="false">
    Whether the switch is disabled.
  </Prop>
  <Prop name="color" type="string">
    Accent color when checked.
  </Prop>
  <Prop name="size" type="'small' | 'medium' | 'large'" default="'medium'">
    Size of the switch.
  </Prop>
  <Prop name="highContrast" type="boolean" default="false">
    Uses high contrast colors.
  </Prop>
  <Prop name="accessibilityLabel" type="string">
    Accessibility label.
  </Prop>
</PropTable>
