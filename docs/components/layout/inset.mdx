# Inset

`Inset` is a utility component for applying consistent padding with optional overflow clipping.

## Import

```tsx
import { Inset } from '@radix-ui/themes-native';
```

## Usage

```tsx
import { Inset, Text } from '@radix-ui/themes-native';

<Inset p={4}>
  <Text>Content with padding</Text>
</Inset>
```

## Props

<PropTable>
  <Prop name="children" type="ReactNode">
    Child components.
  </Prop>
  <Prop name="style" type="StyleProp<ViewStyle>">
    Custom styles.
  </Prop>
  <Prop name="asChild" type="boolean" default="false">
    Merge props onto immediate child.
  </Prop>
  <Prop name="side" type="'all' | 'x' | 'y' | 'top' | 'bottom' | 'left' | 'right'" default="'all'">
    Side(s) to apply padding.
  </Prop>
  <Prop name="p" type="1 | 2 | 3 | 4 | 5 | 6 | 7 | 8 | 9" default="4">
    Padding size from theme spacing scale.
  </Prop>
  <Prop name="clip" type="'overflow' | 'padding-box'">
    Clip overflow content.
  </Prop>
</PropTable>

## Side Values

| Value | Description |
|-------|-------------|
| `all` | All four sides (default) |
| `x` | Horizontal (left and right) |
| `y` | Vertical (top and bottom) |
| `top` | Top only |
| `bottom` | Bottom only |
| `left` | Left only |
| `right` | Right only |

## Examples

### Basic Inset

```tsx
<Inset p={4}>
  <Text>Content with 16px padding on all sides</Text>
</Inset>
```

### Horizontal Padding

```tsx
<Inset side="x" p={4}>
  <Text>Horizontal padding only</Text>
</Inset>
```

### Vertical Padding

```tsx
<Inset side="y" p={3}>
  <Text>Vertical padding only</Text>
</Inset>
```

### With Overflow Clipping

```tsx
<Card radius="large">
  <Inset clip="overflow">
    <Image
      source={{ uri: 'https://example.com/image.jpg' }}
      style={{ width: '100%', height: 150 }}
    />
  </Inset>
  <Inset p={4}>
    <Text>Content below image</Text>
  </Inset>
</Card>
```

### Card with Inset

```tsx
<Card>
  <Inset side="x" p={4}>
    <Flex direction="column" gap={2}>
      <Heading size={3}>Card Title</Heading>
      <Text>Card content with horizontal padding.</Text>
    </Flex>
  </Inset>
</Card>
```

### Top and Bottom

```tsx
<Inset side="top" p={2}>
  <Text>Top padding only</Text>
</Inset>

<Inset side="bottom" p={2}>
  <Text>Bottom padding only</Text>
</Inset>
```
